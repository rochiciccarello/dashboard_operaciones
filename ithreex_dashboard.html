Esquema recomendado PostgreSQL
-- Proyectos
CREATE TABLE proyectos (
  id          SERIAL PRIMARY KEY,
  nombre      TEXT NOT NULL,
  cliente     TEXT,
  estado      TEXT,  -- 'En Curso','Finalizado','Planificado'
  inicio      DATE,
  fin_plan    DATE,
  hs_estimadas NUMERIC,
  presupuesto  NUMERIC,
  clockify_id  TEXT,
  odoo_id      TEXT,
  squad        TEXT
);

-- Horas por colaborador/semana
CREATE TABLE horas_semana (
  id          SERIAL PRIMARY KEY,
  proyecto_id INT REFERENCES proyectos(id),
  colaborador TEXT,
  semana      INT,
  mes         DATE,
  hs_objetivo NUMERIC,
  hs_real     NUMERIC,
  productividad NUMERIC,
  estado_carga TEXT
);

-- Clockify raw
CREATE TABLE clockify_entries (
  id          SERIAL PRIMARY KEY,
  proyecto    TEXT,
  cliente     TEXT,
  descripcion TEXT,
  usuario     TEXT,
  facturable  BOOLEAN,
  fecha       DATE,
  duracion_h  NUMERIC,
  mes         DATE
);

-- Costos por perfil/mes
CREATE TABLE costos_perfil (
  id          SERIAL PRIMARY KEY,
  mes         DATE,
  perfil      TEXT,
  costo_hora  NUMERIC,
  costo_opex  NUMERIC,
  total       NUMERIC,
  costo_venta NUMERIC
);

-- Encuestas satisfacción
CREATE TABLE satisfaccion (
  id          SERIAL PRIMARY KEY,
  proyecto_id INT REFERENCES proyectos(id),
  fecha       DATE,
  score       NUMERIC,
  tipo        TEXT,  -- 'interna','clutch'
  comentarios TEXT
);
